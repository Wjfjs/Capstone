CREATE TABLE TrafficLight (
    TraffID INT auto_increment PRIMARY KEY,
    TrafficLightID INT,  
    Color INT NOT NULL,                     
    Sequence VARCHAR(10) NOT NULL,                          
    ColorTime INT NOT NULL,
    SignalControlNumber INT not null,
    FOREIGN KEY (SignalControlNumber) REFERENCES location(SignalControlNumber)
);

CREATE TABLE location (
    SignalControlNumber INT PRIMARY KEY,
    City VARCHAR(50),
    District VARCHAR(50),
    RouteNumber VARCHAR(10),
    RouteName VARCHAR(100),
    RouteDirection VARCHAR(50),
    Address VARCHAR(200)
);

create table countFirst (
    id int AUTO_INCREMENT PRIMARY KEY,
    date TIMESTAMP,
    count int default 0,
    SignalControlNumber INT not null,
    FOREIGN KEY (SignalControlNumber) REFERENCES TrafficLight(SignalControlNumber)
);

create table countSec (
    id int AUTO_INCREMENT PRIMARY KEY,
    date TIMESTAMP,
    count int default 0,
    SignalControlNumber INT not null,
    FOREIGN KEY (SignalControlNumber) REFERENCES TrafficLight(SignalControlNumber)
);

신호등 TrafficLight
TrafficLightID: 각 신호등의 고유 번호, 기본 키로 설정되어 있으며 자동으로 증가합니다.
Color: 신호등의 색상 (빨강, 초록, 노랑 등).
Sequence: 신호등화 순서, 각 색상이 나타나는 순서를 지정합니다.
ColorTime: 각 색상이 유지되는 시간을 초 단위로 표시합니다.
SignalControlNumber : location 외래키
